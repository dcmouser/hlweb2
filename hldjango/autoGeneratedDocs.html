BUILT-IN FUNCTION DEFINITIONS


_entryApplyOptions

  • Internal function for applying options to entry
  • Usage: _entryApplyOptions(_entry, autoLead="False", time="None", timePos="", sectionColumns="None", leadColumns="None", sectionBreak="None", leadBreak="None", heading="None", toc="None", childPlugins="None", tombstones="None", address="None", copy="None", render="True", blank="False", mStyle="None", sortIndex="-1", continuedFrom="None", deadline="-1")
  • Parameters:
    • _entry: "Object pointer to entry whose options are being set"; required
    • autoLead: "Automatically assign a lead id"; type=AstValBool; default="False"
    • time: "Duration of lead (in minutes)"; type=AstValNumber
    • timePos: "Position to add time instruction"; type=['', 'start', 'end', 'hide']; default=""
    • sectionColumns: "Number of columns in Section main text"; type=AstValNumber
    • leadColumns: "Number of columns in layout for child leads"; type=AstValNumber
    • sectionBreak: "Break style for section"; type=['', 'none', 'before', 'after', 'solo', 'beforeFacing', 'afterFacing', 'soloFacing', 'soloAfterFacing']
    • leadBreak: "Break style for child leads"; type=['', 'none', 'before', 'after', 'solo', 'beforeFacing', 'afterFacing', 'soloFacing', 'soloAfterFacing']
    • heading: "Heading text shown on page (blank for none)"; type=AstValString
    • toc: "Label for table of contents (blank to hide from table of contents)"; type=AstValString
    • childPlugins: "Name of the plugins to run on children (optional)"; type=AstValString
    • tombstones: "Should we show tombstombs between child entries"; type=AstValBool
    • address: "Address line under heading/label (optional); set to 'auto' to auto grab from database or '' blank to mean none"; type=AstValString
    • copy: "Entry id to copy contents from, or 'next' to copy from next lead"; type=AstValString
    • render: "If set false this lead will not render"; type=AstValBool; default="True"
    • blank: "If set true then we will not show the header or add this to toc"; type=AstValBool; default="False"
    • mStyle: "Set to a string keyword that informs mind map drawing (optional)"; type=AstValString
    • sortIndex: "Sort index"; type=AstValNumber; default="-1"
    • continuedFrom: "Add label saying 'continued from this lead"; type=AstValString
    • deadline: "Deadline day for a check"; type=AstValNumber; default="-1"
